04385   {
04386     if(inlen % 2)
04387       BOOST_AFIO_THROW(std::invalid_argument("Input buffer not multiple of two."));
04388     if(outlen < inlen / 2)
04389       BOOST_AFIO_THROW(std::invalid_argument("Output buffer too small."));
04390     bool is_invalid = false;

04400                                                                10,  11,  12,  13,  14,  15};
04401       unsigned char r = 255;
04402       if(c >= 48 && c <= 102)
04403         r = table[c - 48];
04404       if(r == 255)
04405         is_invalid = true;
04406       return r;
04407 #  else  /*_ 1 _*/
04408       if(c >= '0' && c <= '9')
04409         return c - '0';
04410       if(c >= 'a' && c <= 'f')
04411         return c - 'a' + 10;
04412       if(c >= 'A' && c <= 'F')
04413         return c - 'A' + 10;
04414       BOOST_AFIO_THROW(std::invalid_argument("Input is not hexadecimal."));
04415 #  endif  /*_ 1 _*/
04416     };
04417     for(size_t n = 0; n < inlen / 2; n += 4)
04418     {
04419       unsigned char c[8];

04421       c[1] = fromhex(in[n * 2 + 1]);
04422       c[2] = fromhex(in[n * 2 + 2]);
04423       c[3] = fromhex(in[n * 2 + 3]);

04425       c[4] = fromhex(in[n * 2 + 4]);

04428       c[6] = fromhex(in[n * 2 + 6]);
04429       c[7] = fromhex(in[n * 2 + 7]);

04434     {
04435       unsigned char c1 = fromhex(in[n * 2]), c2 = fromhex(in[n * 2 + 1]);
04436       out[n] = (c2 << 4) | c1;
04437     }
04438     if(is_invalid)
04439       BOOST_AFIO_THROW(std::invalid_argument("Input is not hexadecimal."));
04440     return inlen / 2;
04441   }

04658     {

04660       //#pragma omp parallel for reduction(^:ecc)
04661       for(size_t i = 0; i < length; i += sizeof(unit_type))
04662       {

04664         if(!c)                                         // min 1 cycle
04665           continue;
04666         char bitset[bits_per_byte * sizeof(unit_type)];
04667         result_type prefetch[bits_per_byte * sizeof(unit_type)];
04668         // Most compilers will roll this out
04669         for(size_t n = 0; n < bits_per_byte * sizeof(unit_type); n++)  // min 16 cycles
04670         {
04671           bitset[n] = !!(c & ((unit_type) 1 << n));
04672           prefetch[n] = ecc_table[i * bits_per_byte + n];  // min 8 cycles
04673         }
04674         result_type localecc = 0;
04675         for(size_t n = 0; n < bits_per_byte * sizeof(unit_type); n++)
04676         {
04677           if(bitset[n])               // min 8 cycles
04678             localecc ^= prefetch[n];  // min 8 cycles
04679         }
04680         ecc ^= localecc;  // min 1 cycle. Total cycles = min 43 cycles/byte
04681       }

04687     {
04688       result_type length = blocksize * bits_per_byte, eccdiff = good_ecc ^ bad_ecc;
04689       if(_is_single_bit_set(eccdiff))
04690         return (result_type) -1;
04691       for(result_type i = 0, b = 1; i < length; i++, b++)
04692       {
04693         // Skip parity bits
04694         while(_is_single_bit_set(b))
04695           b++;
04696         if(b == eccdiff)
04697           return i;
04698       }
